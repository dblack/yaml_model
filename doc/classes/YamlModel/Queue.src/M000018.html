    def qcreate(action)
      sc = class << self; self; end
      qname = "#{action}_queue"
      ivar = "@#{qname}"

      sc.class_eval do
        define_method(qname) do
          instance_variable_get(ivar) || instance_variable_set(ivar, [])
        end
      end

      sc.class_eval "def \#{action}(&block)\nsend(\"\#{qname}\") << block if block\nend\n"
    end
